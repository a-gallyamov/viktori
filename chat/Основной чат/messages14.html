/*
* PID.c
*
* Created: 15.11.2021 23:04:19
* Author: dima
*/

#include "PID.h"
/*
input - значение с датчика, тоесть то что мы измерили например с термистора
setpoint - установка, например та температура которую хотим получить
Kp - пропорциональный коэффициент
Ki - интегральный коэффициент
Kd - дифференциальный коэффициент
Dt - период времени через который вызываем функцию в секундах. Предположим мы вызываем данную функцию через таймер раз
в 1 сек., тогда вместо Dt в программе вставим 1, если функцию вызываем раз в 20 мсек, тогда Dt укажем 0.02

Reset - сброс пид регулятора. Если RESET - 1 сброс, 0 - сброса нет, обычный рабочий режим.
*/
//управление зарядным током батареи
int16_t PID_Regulator_I_Mosfet(int16_t input, int16_t setpoint, uint8_t reset)
{
float P = 0;
static float I = 0;
float D = 0;

static float last_error = 0;
static float I_last_error;
float Kp = 4, Ki = 50, Kd = 0, Dt = 0.1;
int16_t min_out = 0, max_out = 1023;//ограничиваем выходной сигнал ПИД регулятора возможный диапазон от -32768 до 32767(а выбираем что нужно)
int16_t pid_out;

if (reset)
{
I = 0;
last_error = 0;
I_last_error = 0;
return 0;
}

P = setpoint - input;
I = I + (P * Dt);
D = (P - last_error) / Dt;
last_error = P;

pid_out = (P * Kp + I * Ki + D * Kd);

if (pid_out >= min_out && pid_out <= max_out)//
{
I_last_error = I;
return pid_out;
}
else if(pid_out < min_out)//out < min
{
I = I_last_error;
return min_out;
}
else//out > max
{
I = I_last_error;
return max_out;
}
}

/*
пример вызова
int16_t out;

out = PID_Regulator(input, setpoint,0);
*/

1. Классическая тиристорная. Работает только и только на сетевом трансформаторе другого не дано. Тиристорные зарядки заряжают высокотоковыми короткими импульсами частотой 50 или 100Гц.

3. Линейные зарядки с использованием DC/DC преобразователя, кпд выше так как нагрев преобразователя меньше. Ток на выходе так же линейный.

16 битный ЦАП, управляемый по SPI.

void Send_Dac(uint16_t data)
{
#ifdef DAC8830
PORT_SPI &= ~(1 << DAC_CP);

SPDR = (uint8_t)(data >> 8);
while(!(SPSR & (1<<SPIF)));

SPDR = (uint8_t)(data & 0xff);
while(!(SPSR & (1<<SPIF)));

PORT_SPI |= (1 << DAC_CP);
#endif

#ifdef MCP4725

I2C_SendWordByADDR((data & 0x0FFF), ADDR_MCP4725);

#endif
}


void init_spi(void)
{
//Управление DAC
DDR_SPI |= (1 << DAC_CP);// output
PORT_SPI |= (1 << DAC_CP);

DDR_SPI |= ((1 << SPI_CLK) | (1 << SPI_DS)); // output

SPCR = ((1<<SPE) | (1<<MSTR) ); // enable SPI, MASTER mode, prescaler 4
//SPCR=(1<<SPE)|(1<<MSTR)|(1<<SPR1)|(1<<SPR0); // enable SPI, MASTER mode, prescaler 128
//SPCR = ((1<<SPE) | (1<<MSTR) | (1<<SPR0)); // enable SPI, MASTER mode, prescaler 16
//SPCR = ((1<<SPE) | (1<<MSTR) | (1<<SPR1)); // enable SPI, MASTER mode, prescaler 64
SPSR = 0;
}

В микроконтроллерах AVR есть три вида памяти, еепром, флеш и озу. В общем и целом еепром и флеш это по факту обе памяти еепром, они обе энергонезависимые и после отключения памяти данные оттуда не удаляются. Отличие только в способе их использования. Во флеш загоняется наша программа которую мы написали и эти данные во время работы уже изменить нельзя(на самом деле можно но с определенными ограничениями и это редко используется). Данные в еепром можно менять во время исполнения программы как хотим и когда хотим. И эти обе памяти можно использовать под завязку до последнего байта и бита, с этим нет никаких проблем и не важно что у нас подключено дисплей 1602 или программируемая промышленная графическая панель WEINTEK на 2000-тыс страниц. А вот с ОЗУ тут надо осторожно. Мы не можем использовать всю ОЗУ в проекте даже если хочется. В ОЗУ храним создаваемые нами переменные, но и сам микроконтроллер использует тоже ОЗУ для своих нужд когда заходит в прерывания. И вот если свободного ОЗУ мы оставили мало, то произойдет переполнение СТЕКА и тогда все, будет ахтунг и вся логика программы рухнет, навернется все и полностью. Можно сказать что микроконтроллер завис, спасти может из этой ситуации только сторожевой таймер (ватчдог) и то если он был заранее включен. Ватчдог сбросит программу вернув ее на нулевой вектор(это самый первый байт во флеш (это при условии что не используется загрузчик, если используется то по другому вектору)). так вот сколько чип будет использовать озу памяти когда заходит в прерывания мы не знаем, хотя узнать можно если открыть код через ассемблер (эта фишка не доступна в среде ардуино IDE, а вот в среде MicrochipStudio это возможно). ОЗУ используется так же и в заходе в функции и при их выполнении но все надо смотреть отдельно по коду. Так вот использование графического дисплея в первую очередь пожирается память флеш, а не ОЗУ и пожирается просто немеряно(память пожирают функции вывода данных на дисплей, шрифты и прочая шелуха). При этом остается мало памяти для основной логики программы. Мало памяти ? Используйте LCD1602 , для данного проекта графический дисплей просто избыточен, так побаловаться. Надо экономить память? тогда пишите программу в среде разработки MicrochipStudio, там есть возможность включить максимальную оптимизацию кода, что сразу сэкономит пару килобайт памяти при компиляции и так же не будет ненужной шелухи (программного кода) которую создает среда ардуино по умолчанию. Это еще минус килобайт.
